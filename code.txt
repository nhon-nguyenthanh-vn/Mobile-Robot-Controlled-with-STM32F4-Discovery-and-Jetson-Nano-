void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart)
{
	if(huart->Instance==huart1.Instance)
	{	/**/
		/*--------NHAN DU LIEU--------*/
		d[0]=d[1];d[1]=d[2];d[2]=d[3];d[3]=d[4];
		theta[0]=theta[1];theta[1]=theta[2];theta[2]=theta[3];theta[3]=theta[4];
		/*KHOANG CACH*/
		d[4] = ((float)receive_data[0]-48)*10 + ((float)receive_data[1]-48) 
				 + ((float)receive_data[3]-48)/10 + ((float)receive_data[4]-48)/100; 
		/*GOC*/
		theta[4] = ((float)receive_data[6]-48)*10 + ((float)receive_data[7]-48) 
						 + ((float)receive_data[9]-48)/10 + ((float)receive_data[10]-48)/100;
		/*KIEM TRA DAU*/
		if((receive_data[5]-48)==1)	
			theta[4] = -theta[4];
		/*--------------------------*/
		/*HIEN LCD*/
		for (i=0;i<5;i++)
		{
		dout[i] = receive_data[i];
		wout[i] = receive_data[i+6];
		}
		lcd_goto_XY(1,0);
		lcd_send_string(dout);
		lcd_send_string(" cm");
		lcd_goto_XY(2,0);
		if(receive_data[5] - 48)
			lcd_send_string("-");
		else 
			lcd_send_string(" ");
		lcd_send_string(wout);
		lcd_send_string(" do");
	/*------------------*/
		
		
	/*------PID VAN TOC DAI------*/
	e_d_2 = e_d_1;
	e_d_1 = e_d; 
	e_d = d[4] - d0;
	if(e_d<2 && e_d>-2)
			e_d = 0;
	v_out_1 = v_out;
	v_out = v_out_1 + Kp_d*(e_d - e_d_1) + Ki_d*Tu/2*(e_d-e_d_1) + Kd_d*Tu*(e_d - 2*e_d_1 + e_d_2);//Van toc dai
	/*-------------------------------*/
	
	/*------PID VAN TOC GOC------*/
	w = theta[4];//Sai so Goc
	e_w_2 = e_w_1;
	e_w_1 = e_w; 
	e_w = w - 0;
		if(e_w < 2 && e_w > -2)
			e_w = 0;
	w_out_1 = w_out;
	w_out = w_out_1 + Kp_w*(e_w - e_w_1) + Ki_w*Tu/2*(e_w-e_w_1) + Kd_w*Tu*(e_w - 2*e_w_1 + e_w_2);// Van toc goc
	/*-------------------------------*/
	
	/*TINH WL + WR*/
	wlf0 = (v_out - b*w_out)/R;//(rad/s)
	wlf0 = wlf0*ppr/(100*2*3.14);//(xung/10ms)
	wrt0 = (v_out + b*w_out)/R;	
	wrt0 = wrt0*ppr/(2*3.14*100);
	if(wlf0 > w_lm)
			wlf0 = w_lm;
	if(wlf0 < -w_lm)
			wlf0 = -w_lm;
	if(wrt0 > w_lm)
			wrt0 = w_lm;
	if(wrt0 < -w_lm)
			wrt0 = -w_lm;
	/*-----*/
	}
}
void HAL_TIM_PeriodElapsedCallback(TIM_HandleTypeDef *htim)
{
	if(htim->Instance == htim3.Instance)
		{
			/*CHON VI TRI CHAN ENC*
			end_lf = __HAL_TIM_GET_COUNTER(&htim2);//cua thang right
				end_rt = __HAL_TIM_GET_COUNTER(&htim5);//cua thang left
			wlf = (end_lf - pre_end_lf);
			wrt = (end_rt - pre_end_rt);
	pre_end_lf = end_lf;
			pre_end_rt= end_rt;
		
			
		__HAL_TIM_SET_COMPARE(&htim4,TIM_CHANNEL_3,30);//cua thang left
		__HAL_TIM_SET_COMPARE(&htim4,TIM_CHANNEL_4,0);
			__HAL_TIM_SET_COMPARE(&htim4,TIM_CHANNEL_2,0);//cua thang right
			__HAL_TIM_SET_COMPARE(&htim4,TIM_CHANNEL_1,0);
			/*----------------*/
			
			/*------PID W LEFT------*/
	end_lf = __HAL_TIM_GET_COUNTER(&htim2);
	wlf = (end_lf - pre_end_lf);
	pre_end_lf = end_lf;
	e_wlf_2 = e_wlf_1;
	e_wlf_1 = e_wlf; 
	e_wlf = wlf0 - wlf;
	u_lf_out_1 = u_lf_out;
	u_lf_out = u_lf_out_1 + Kp_wlf*(e_wlf - e_wlf_1) + Ki_wlf*Ts/2*(e_wlf-e_wlf_1) + Kd_wlf*Ts*(e_wlf - 2*e_wlf_1 + e_wlf_2);//Van toc dai
	duty_lf = u_lf_out/12*255;
	/*------------------------------*/
			
		/*------PID W RIGHT------*/
	end_rt = __HAL_TIM_GET_COUNTER(&htim5);
	wrt = (end_rt - pre_end_rt);
	pre_end_rt = end_rt;
	e_wrt_2 = e_wrt_1;
	e_wrt_1 = e_wrt; 
	e_wrt = wrt0 - wrt;
	u_rt_out_1 = u_rt_out;
	u_rt_out = u_rt_out_1 + Kp_wrt*(e_wrt - e_wrt_1) + Ki_wrt*Ts/2*(e_wrt-e_wrt_1) + Kd_wrt*Ts*(e_wrt - 2*e_wrt_1 + e_wrt_2);//Van toc dai
	duty_rt = u_rt_out/12*255;
	/*------------------------------*/
	
	/*--------XUAT PWM W LEFT------------*
	if(duty_lf > 0)
	{
			duty_lf += 0;
	if(duty_lf > 255)
			duty_lf = 255;
	__HAL_TIM_SET_COMPARE(&htim4,TIM_CHANNEL_4,0);
	__HAL_TIM_SET_COMPARE(&htim4,TIM_CHANNEL_3,duty_lf);
	}
	else
	{
					duty_lf -= 0;
			if(duty_lf < (-255))
					duty_lf = (-255);
	__HAL_TIM_SET_COMPARE(&htim4,TIM_CHANNEL_3,0);
	__HAL_TIM_SET_COMPARE(&htim4,TIM_CHANNEL_4,-duty_lf);
	}
	/*------------------------*/
	
	/*--------XUAT PWM W RIGHT------------*
	if(duty_rt > 0)
	{
	if(duty_rt > 255)
			duty_rt = 255;
	__HAL_TIM_SET_COMPARE(&htim4,TIM_CHANNEL_1,0);
	__HAL_TIM_SET_COMPARE(&htim4,TIM_CHANNEL_2,duty_rt);
	}
	else
	{
			if(duty_rt < (-255))
					duty_rt = (-255);
	__HAL_TIM_SET_COMPARE(&htim4,TIM_CHANNEL_2,0);
	__HAL_TIM_SET_COMPARE(&htim4,TIM_CHANNEL_1,-duty_rt);
	}
	/*------------------------*/			
		}
	}
